import xgboost as xgb
from xgboost import XGBRegressor
import sklearn
import pandas

data = pandas.read_csv("prostate54-2.csv")
y_train = data["AGE"]
X_train = data[["PSA", "VOL"]]
# create model instance
bst = XGBRegressor(n_estimators=8, max_depth=5, learning_rate=0.3, objective='reg:squarederror', seed=847020, booster="gbtree")
# fit model
bst.fit(X_train, y_train)
bst.save_model("boosterBytesProstate.bin")
